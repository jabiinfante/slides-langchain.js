const stream = await agent.stream(
  { messages: [{ role: "user", content: "¿Qué tiempo hace en Barcelona?" }] },
  { configurable: { thread_id: "thread-1" } }
);

for await (const event of stream) {
  console.log("---");
  console.log(event);
  // Muestra cada paso:
  // 1. Decisión del LLM
  // 2. Ejecución de la tool
  // 3. Resultado de la tool
  // 4. Respuesta final
}
