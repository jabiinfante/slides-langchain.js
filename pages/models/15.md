# Ejemplo: Agente con Tavily Search

<div class="text-xs mt-2">

```typescript
import { ChatOpenAI } from "@langchain/openai";
import { TavilySearchResults } from "@langchain/community/tools/tavily_search";
import { createReactAgent } from "@langchain/langgraph/prebuilt";

const model = new ChatOpenAI({ model: "gpt-4o-mini" });
const searchTool = new TavilySearchResults({ apiKey: process.env.TAVILY_API_KEY, maxResults: 5 });
const agent = createReactAgent({ llm: model, tools: [searchTool] });

const result = await agent.invoke({
  messages: [{ role: "user", content: "Â¿Cuales son las ultimas noticias sobre IA?" }],
});
```
